92 template <typename T>
93 bool BoundedQueue<T>::Init(uint64_t size, WaitStrategy* strategy) {
94  // Head and tail each occupy a space
95  pool_size_ = size + 2;
96  pool_ = reinterpret_cast<T*>(std::calloc(pool_size_, sizeof(T)));
97  if (pool_ == nullptr) {
98  return false;
99  }
100  for (uint64_t i = 0; i < pool_size_; ++i) {
101  new (&(pool_[i])) T();
102  }
103  wait_strategy_.reset(strategy);
104  return true;
105 }

107 template <typename T>
108 bool BoundedQueue<T>::Enqueue(const T& element) {
109  uint64_t new_tail = 0;
110  uint64_t old_commit = 0;
111  uint64_t old_tail = tail_.load(std::memory_order_acquire);
112  do {
113  new_tail = old_tail + 1;
114  if (GetIndex(new_tail) == GetIndex(head_.load(std::memory_order_acquire))) {
115  return false;
116  }
117  } while (!tail_.compare_exchange_weak(old_tail, new_tail,
118  std::memory_order_acq_rel,
119  std::memory_order_relaxed));
120  pool_[GetIndex(old_tail)] = element;
121  do {
122  old_commit = old_tail;
123  } while (unlikely(!commit_.compare_exchange_weak(old_commit, new_tail,
124  std::memory_order_acq_rel,
125  std::memory_order_relaxed)));
126  wait_strategy_->NotifyOne();
127  return true;
128 }

153 template <typename T>
154 bool BoundedQueue<T>::Dequeue(T* element) {
155  uint64_t new_head = 0;
156  uint64_t old_head = head_.load(std::memory_order_acquire);
157  do {
158  new_head = old_head + 1;
159  if (new_head == commit_.load(std::memory_order_acquire)) {
160  return false;
161  }
162  *element = pool_[GetIndex(new_head)];
163  } while (!head_.compare_exchange_weak(old_head, new_head,
164  std::memory_order_acq_rel,
165  std::memory_order_relaxed));
166  return true;
167 }

110 template <typename Request, typename Response>
111 bool Client<Request, Response>::Init() {
112  proto::RoleAttributes role;
113  role.set_node_name(node_name_);
114  role.set_channel_name(request_channel_);
115  auto channel_id = common::GlobalData::RegisterChannel(request_channel_);
116  role.set_channel_id(channel_id);
117  role.mutable_qos_profile()->CopyFrom(
118  transport::QosProfileConf::QOS_PROFILE_SERVICES_DEFAULT);
119  auto transport = transport::Transport::Instance();
120  request_transmitter_ =
121  transport->CreateTransmitter<Request>(role, proto::OptionalMode::RTPS);
122  if (request_transmitter_ == nullptr) {
123  AERROR << "Create request pub failed.";
124  return false;
125  }
126  writer_id_ = request_transmitter_->id();
127 
128  response_callback_ =
129  std::bind(&Client<Request, Response>::HandleResponse, this,
130  std::placeholders::_1, std::placeholders::_2);
131 
132  role.set_channel_name(response_channel_);
133  channel_id = common::GlobalData::RegisterChannel(response_channel_);
134  role.set_channel_id(channel_id);
135  response_receiver_ = transport->CreateReceiver<Response>(
136  role,
137  [=](const std::shared_ptr<Response>& request,
138  const transport::MessageInfo& message_info,
139  const proto::RoleAttributes& reader_attr) {
140  (void)message_info;
141  (void)reader_attr;
142  response_callback_(request, message_info);
143  },
144  proto::OptionalMode::RTPS);
145  if (response_receiver_ == nullptr) {
146  AERROR << "Create response sub failed.";
147  request_transmitter_.reset();
148  return false;
149  }
150  return true;
151 }

154 typename Client<Request, Response>::SharedResponse
155 Client<Request, Response>::SendRequest(SharedRequest request,
156  const std::chrono::seconds& timeout_s) {
157  if (!IsInit()) {
158  return nullptr;

160  auto future = AsyncSendRequest(request);

162  return nullptr;
163  }
164  auto status = future.wait_for(timeout_s);

196 template <typename Request, typename Response>
197 typename Client<Request, Response>::SharedFuture
198 Client<Request, Response>::AsyncSendRequest(SharedRequest request,
199  CallbackType&& cb) {
200  if (IsInit()) {
201  std::lock_guard<std::mutex> lock(pending_requests_mutex_);
202  sequence_number_++;
203  transport::MessageInfo info(writer_id_, sequence_number_, writer_id_);
204  request_transmitter_->Transmit(request, info);
205  SharedPromise call_promise = std::make_shared<Promise>();
206  SharedFuture f(call_promise->get_future());
207  pending_requests_[info.seq_num()] =
208  std::make_tuple(call_promise, std::forward<CallbackType>(cb), f);
209  return f;
210  } else {
211  return std::shared_future<std::shared_ptr<Response>>();
212  }
213 }

220 template <typename Request, typename Response>
221 void Client<Request, Response>::HandleResponse(
222  const std::shared_ptr<Response>& response,
223  const transport::MessageInfo& request_header) {
224  ADEBUG << "client recv response.";
225  std::lock_guard<std::mutex> lock(pending_requests_mutex_);
226  if (request_header.spare_id() != writer_id_) {
227  return;
228  }
229  uint64_t sequence_number = request_header.seq_num();
230  if (this->pending_requests_.count(sequence_number) == 0) {
231  return;
232  }
233  auto tuple = this->pending_requests_[sequence_number];
234  auto call_promise = std::get<0>(tuple);
235  auto callback = std::get<1>(tuple);
236  auto future = std::get<2>(tuple);
237  this->pending_requests_.erase(sequence_number);
238  call_promise->set_value(response);
239  callback(future);
240 }

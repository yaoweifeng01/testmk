111 template <typename MessageT>
112 auto Node::CreateReader(const proto::RoleAttributes& role_attr,
113  const CallbackFunc<MessageT>& reader_func)
114  -> std::shared_ptr<Reader<MessageT>> {
115  std::lock_guard<std::mutex> lg(readers_mutex_);
116  if (readers_.find(role_attr.channel_name()) != readers_.end()) {
117  AWARN << "Failed to create reader: reader with the same channel already "
118  "exists.";
119  return nullptr;
120  }
121  auto reader = node_channel_impl_->template CreateReader<MessageT>(
122  role_attr, reader_func);
123  if (reader != nullptr) {
124  readers_.emplace(std::make_pair(role_attr.channel_name(), reader));
125  }
126  return reader;
127 }

81 template <typename M>
82 auto Transport::CreateTransmitter(const RoleAttributes& attr,
83  const OptionalMode& mode) ->
84  typename std::shared_ptr<Transmitter<M>> {
85  if (is_shutdown_.load()) {
86  AINFO << "transport has been shut down.";
87  return nullptr;
88  }
89 
90  std::shared_ptr<Transmitter<M>> transmitter = nullptr;
91  RoleAttributes modified_attr = attr;
92  if (!modified_attr.has_qos_profile()) {
93  modified_attr.mutable_qos_profile()->CopyFrom(
94  QosProfileConf::QOS_PROFILE_DEFAULT);
95  }
96 
97  switch (mode) {
98  case OptionalMode::INTRA:
99  transmitter = std::make_shared<IntraTransmitter<M>>(modified_attr);
100  break;
101 
102  case OptionalMode::SHM:
103  transmitter = std::make_shared<ShmTransmitter<M>>(modified_attr);
104  break;
105 
106  case OptionalMode::RTPS:
107  transmitter =
108  std::make_shared<RtpsTransmitter<M>>(modified_attr, participant());
109  break;
110 
111  default:
112  transmitter =
113  std::make_shared<HybridTransmitter<M>>(modified_attr, participant());
114  break;
115  }
116 
117  RETURN_VAL_IF_NULL(transmitter, nullptr);
118  if (mode != OptionalMode::HYBRID) {
119  transmitter->Enable();
120  }
121  return transmitter;
122 }

93 template <typename MessageT>
94 auto ReceiverManager<MessageT>::GetReceiver(
95  const proto::RoleAttributes& role_attr) ->
96  typename std::shared_ptr<transport::Receiver<MessageT>> {
97  std::lock_guard<std::mutex> lock(receiver_map_mutex_);
98  // because multi reader for one channel will write datacache multi times,
99  // so reader for datacache we use map to keep one instance for per channel
100  const std::string& channel_name = role_attr.channel_name();
101  if (receiver_map_.count(channel_name) == 0) {
102  receiver_map_[channel_name] =
103  transport::Transport::Instance()->CreateReceiver<MessageT>(
104  role_attr, [](const std::shared_ptr<MessageT>& msg,
105  const transport::MessageInfo& msg_info,
106  const proto::RoleAttributes& reader_attr) {
107  (void)msg_info;
108  (void)reader_attr;
109  PerfEventCache::Instance()->AddTransportEvent(
110  TransPerf::TRANS_TO, reader_attr.channel_id(),
111  msg_info.seq_num());
112  data::DataDispatcher<MessageT>::Instance()->Dispatch(
113  reader_attr.channel_id(), msg);
114  PerfEventCache::Instance()->AddTransportEvent(
115  TransPerf::WRITE_NOTIFY, reader_attr.channel_id(),
116  msg_info.seq_num());
117  });
118  }
119  return receiver_map_[channel_name];
120 }

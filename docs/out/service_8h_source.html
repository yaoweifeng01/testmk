110 template <typename Request, typename Response>
111 void Service<Request, Response>::Process() {
112  while (!cyber::IsShutdown()) {
113  std::unique_lock<std::mutex> ul(queue_mutex_);
114  condition_.wait(ul, [this]() { return !inited_ || !this->tasks_.empty(); });
115  if (!inited_) {
116  break;
117  }
118  if (!tasks_.empty()) {
119  auto task = tasks_.front();
120  tasks_.pop_front();
121  ul.unlock();
122  task();
123  }
124  }
125 }

147  request_callback_ =

149  std::placeholders::_1, std::placeholders::_2);

152  channel_id = common::GlobalData::RegisterChannel(request_channel_);
153  role.set_channel_id(channel_id);
154  request_receiver_ = transport->CreateReceiver<Request>(
155  role,
156  [=](const std::shared_ptr<Request>& request,
157  const transport::MessageInfo& message_info,
158  const proto::RoleAttributes& reader_attr) {
159  (void)reader_attr;
160  (void)reader_attr;
161  auto task = [this, request, message_info]() {
162  this->HandleRequest(request, message_info);
163  };
164  Enqueue(std::move(task));
165  },
166  proto::OptionalMode::RTPS);
167  inited_ = true;
168  thread_ = std::thread(&Service<Request, Response>::Process, this);
169  if (request_receiver_ == nullptr) {
170  AERROR << " Create request sub failed." << request_channel_;
171  response_transmitter_.reset();
172  return false;
173  }
174  return true;
175 }

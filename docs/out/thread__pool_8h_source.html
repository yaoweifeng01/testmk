52 inline ThreadPool::ThreadPool(std::size_t threads, std::size_t max_task_num)
53  : stop_(false) {
54  if (!task_queue_.Init(max_task_num, new BlockWaitStrategy())) {
55  throw std::runtime_error("Task queue init failed.");
56  }
57  for (size_t i = 0; i < threads; ++i) {
58  workers_.emplace_back([this] {
59  while (!stop_) {
60  std::function<void()> task;
61  if (task_queue_.WaitDequeue(&task)) {
62  task();
63  }
64  }
65  });
66  }
67 }

70 template <typename F, typename... Args>
71 auto ThreadPool::Enqueue(F&& f, Args&&... args)
72  -> std::future<typename std::result_of<F(Args...)>::type> {
73  using return_type = typename std::result_of<F(Args...)>::type;
74 
75  auto task = std::make_shared<std::packaged_task<return_type()>>(
76  std::bind(std::forward<F>(f), std::forward<Args>(args)...));
77 
78  std::future<return_type> res = task->get_future();
79 
80  // don't allow enqueueing after stopping the pool
81  if (stop_) {
82  return std::future<return_type>();
83  }
84  task_queue_.Enqueue([task]() { (*task)(); });
85  return res;
86 };

64 inline void ReentrantRWLock::ReadLock() {
65  if (write_thread_id_ == std::this_thread::get_id()) {
66  return;
67  }
68 
69  uint32_t retry_times = 0;
70  int32_t lock_num = lock_num_.load(std::memory_order_acquire);
71  if (write_first_) {
72  do {
73  while (lock_num < RW_LOCK_FREE ||
74  write_lock_wait_num_.load(std::memory_order_acquire) > 0) {
75  if (++retry_times == MAX_RETRY_TIMES) {
76  // saving cpu
77  std::this_thread::yield();
78  retry_times = 0;
79  }
80  lock_num = lock_num_.load(std::memory_order_acquire);
81  }
82  } while (!lock_num_.compare_exchange_weak(lock_num, lock_num + 1,
83  std::memory_order_acq_rel,
84  std::memory_order_relaxed));
85  } else {
86  do {
87  while (lock_num < RW_LOCK_FREE) {
88  if (++retry_times == MAX_RETRY_TIMES) {
89  // saving cpu
90  std::this_thread::yield();
91  retry_times = 0;
92  }
93  lock_num = lock_num_.load(std::memory_order_acquire);
94  }
95  } while (!lock_num_.compare_exchange_weak(lock_num, lock_num + 1,
96  std::memory_order_acq_rel,
97  std::memory_order_relaxed));
98  }
99 }

101 inline void ReentrantRWLock::WriteLock() {
102  auto this_thread_id = std::this_thread::get_id();
103  if (write_thread_id_ == this_thread_id) {
104  lock_num_.fetch_sub(1);
105  return;
106  }
107  int32_t rw_lock_free = RW_LOCK_FREE;
108  uint32_t retry_times = 0;
109  write_lock_wait_num_.fetch_add(1);
110  while (!lock_num_.compare_exchange_weak(rw_lock_free, WRITE_EXCLUSIVE,
111  std::memory_order_acq_rel,
112  std::memory_order_relaxed)) {
113  // rw_lock_free will change after CAS fail, so init agin
114  rw_lock_free = RW_LOCK_FREE;
115  if (++retry_times == MAX_RETRY_TIMES) {
116  // saving cpu
117  std::this_thread::yield();
118  retry_times = 0;
119  }
120  }
121  write_thread_id_ = this_thread_id;
122  write_lock_wait_num_.fetch_sub(1);
123 }

125 inline void ReentrantRWLock::ReadUnlock() {

127  return;
128  }
129  lock_num_.fetch_sub(1);
130 }

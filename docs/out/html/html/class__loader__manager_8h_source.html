62 template <typename Base>
63 std::shared_ptr<Base> ClassLoaderManager::CreateClassObj(
64  const std::string& class_name) {
65  std::vector<ClassLoader*> class_loaders = GetAllValidClassLoaders();
66  for (auto class_loader : class_loaders) {
67  if (class_loader->IsClassValid<Base>(class_name)) {
68  return (class_loader->CreateClassObj<Base>(class_name));
69  }
70  }
71  AERROR << "Invalid class name: " << class_name;
72  return std::shared_ptr<Base>();
73 }

88 template <typename Base>
89 bool ClassLoaderManager::IsClassValid(const std::string& class_name) {
90  std::vector<std::string> valid_classes = GetValidClassNames<Base>();
91  return (valid_classes.end() !=
92  std::find(valid_classes.begin(), valid_classes.end(), class_name));

95 template <typename Base>
96 std::vector<std::string> ClassLoaderManager::GetValidClassNames() {
97  std::vector<std::string> valid_classes;
98  for (auto class_loader : GetAllValidClassLoaders()) {
99  std::vector<std::string> class_loaders =
100  class_loader->GetValidClassNames<Base>();
101  valid_classes.insert(valid_classes.end(), class_loaders.begin(),
102  class_loaders.end());
103  }

55  inline void add(const SingleMessage& message) {
56  std::lock_guard<std::mutex> lock(mutex_);
57  SingleMessage* p_message = body_.add_messages();
58  *p_message = message;
59  if (0 == header_.begin_time()) {
60  header_.set_begin_time(message.time());
61  }
62  if (header_.begin_time() > message.time()) {
63  header_.set_begin_time(message.time());
64  }
65  if (header_.end_time() < message.time()) {
66  header_.set_end_time(message.time());
67  }
68  header_.set_message_number(header_.message_number() + 1);
69  header_.set_raw_size(header_.raw_size() + message.content().size());
70  }

105 template <typename T>
106 bool RecordFileWriter::WriteSection(const T& message) {
107  SectionType type;
108  if (std::is_same<T, ChunkHeader>::value) {
109  type = SectionType::SECTION_CHUNK_HEADER;
110  } else if (std::is_same<T, ChunkBody>::value) {
111  type = SectionType::SECTION_CHUNK_BODY;
112  } else if (std::is_same<T, Channel>::value) {
113  type = SectionType::SECTION_CHANNEL;
114  } else if (std::is_same<T, Header>::value) {
115  type = SectionType::SECTION_HEADER;
116  if (!SetPosition(0)) {
117  AERROR << "Jump to position #0 failed";
118  return false;
119  }
120  } else if (std::is_same<T, Index>::value) {
121  type = SectionType::SECTION_INDEX;
122  } else {
123  AERROR << "Do not support this template typename.";
124  return false;
125  }
126  Section section = {type, message.ByteSize()};
127  ssize_t count = write(fd_, &section, sizeof(section));
128  if (count < 0) {
129  AERROR << "Write fd failed, fd: " << fd_ << ", errno: " << errno;
130  return false;
131  }
132  if (count != sizeof(section)) {
133  AERROR << "Write fd failed, fd: " << fd_
134  << ", expect count: " << sizeof(section)
135  << ", actual count: " << count;
136  return false;
137  }
138  ZeroCopyOutputStream* raw_output = new FileOutputStream(fd_);
139  message.SerializeToZeroCopyStream(raw_output);
140  delete raw_output;
141  if (type == SectionType::SECTION_HEADER) {
142  static char blank[HEADER_LENGTH] = {'0'};
143  count = write(fd_, &blank, HEADER_LENGTH - message.ByteSize());
144  if (count < 0) {
145  AERROR << "Write fd failed, fd: " << fd_ << ", errno: " << errno;
146  return false;
147  }
148  if (count != HEADER_LENGTH - message.ByteSize()) {
149  AERROR << "Write fd failed, fd: " << fd_
150  << ", expect count: " << sizeof(section)
151  << ", actual count: " << count;
152  return false;
153  }
154  }
155  header_.set_size(CurrentPosition());
156  return true;
157 }

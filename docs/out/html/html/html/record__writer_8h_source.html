121 template <typename MessageT>
122 bool RecordWriter::WriteMessage(const std::string& channel_name,
123  const MessageT& message,
124  const uint64_t time_nanosec,
125  const std::string& proto_desc) {
126  const std::string& message_type = GetMessageType(channel_name);
127  if (message_type.empty()) {
128  if (!WriteChannel(channel_name, MessageT::descriptor()->full_name(),
129  proto_desc)) {
130  AERROR << "Failed to write meta data to channel [" << channel_name
131  << "].";
132  return false;
133  }
134  } else {
135  if (MessageT::descriptor()->full_name() != message_type) {
136  AERROR << "Message type is invalid, expect: " << message_type
137  << ", actual: " << MessageT::descriptor()->full_name();
138  return false;
139  }
140  }
141  std::string content("");
142  if (!message.SerializeToString(&content)) {
143  AERROR << "Failed to serialize message, channel: " << channel_name;
144  return false;
145  }
146  return WriteMessage(channel_name, content, time_nanosec);
147 }

168 typename std::enable_if<HasParseFromArray<T>::value, bool>::type ParseFromHC(
169  const void* data, int size, T* message) {
170  const auto header_size = sizeof(MessageHeader);
171  RETURN_VAL_IF(size < header_size, false);
172  const MessageHeader* header = static_cast<const MessageHeader*>(data);
173  RETURN_VAL_IF((size - header_size) < header->content_size(), false);
174  SetTypeName(header->msg_type(), message);
175  return message->ParseFromArray(
176  static_cast<const void*>(static_cast<const char*>(data) + header_size),
177  header->content_size());
178 }

210 template <typename T>
211 typename std::enable_if<HasSerializeToArray<T>::value, bool>::type
212 SerializeToHC(const T& message, void* data, int size) {
213  int msg_size = ByteSize(message);
214  if (msg_size < 0) {
215  return false;
216  }
217  const std::string& type_name = MessageType(message);
218  MessageHeader header;
219  header.set_msg_type(type_name.data(), type_name.size());
220  header.set_content_size(msg_size);
221  char* ptr = reinterpret_cast<char*>(data);
222  memcpy(ptr, static_cast<const void*>(&header), sizeof(header));
223  ptr += sizeof(header);
224  int left_size = size - static_cast<int>(sizeof(header));
225  return SerializeToArray(message, reinterpret_cast<void*>(ptr), left_size);
226 }

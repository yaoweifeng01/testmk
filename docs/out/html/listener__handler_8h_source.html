109 template <typename MessageT>
110 void ListenerHandler<MessageT>::Connect(uint64_t self_id, uint64_t oppo_id,
111  const Listener& listener) {
112  WriteLockGuard<AtomicRWLock> lock(rw_lock_);
113  if (signals_.find(oppo_id) == signals_.end()) {
114  signals_[oppo_id] = std::make_shared<MessageSignal>();
115  }
116 
117  auto connection = signals_[oppo_id]->Connect(listener);
118  if (!connection.IsConnected()) {
119  AWARN << oppo_id << " " << self_id << "connect failed!";
120  return;
121  }
122 
123  if (signals_conns_.find(oppo_id) == signals_conns_.end()) {
124  signals_conns_[oppo_id] = ConnectionMap();
125  }
126 
127  signals_conns_[oppo_id][self_id] = connection;
128 }

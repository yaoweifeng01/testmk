130  std::string read(bool wait = false) {
131  std::string msg("");
132  std::unique_lock<std::mutex> ul(msg_lock_);
133  if (!cache_.empty()) {
134  msg = std::move(cache_.front());
135  cache_.pop_front();
136  }
137 
138  if (!wait) {
139  return msg;
140  }
141 
142  msg_cond_.wait(ul, [this] { return !this->cache_.empty(); });
143  if (!cache_.empty()) {

145  cache_.pop_front();
146  }

127 template <typename MessageT>
128 auto NodeChannelImpl::CreateWriter(const proto::RoleAttributes& role_attr)
129  -> std::shared_ptr<Writer<MessageT>> {
130  if (!role_attr.has_channel_name() || role_attr.channel_name().empty()) {
131  AERROR << "Can't create a writer with empty channel name!";
132  return nullptr;
133  }
134  proto::RoleAttributes new_attr(role_attr);
135  FillInAttr<MessageT>(&new_attr);
136 
137  std::shared_ptr<Writer<MessageT>> writer_ptr = nullptr;
138  if (!is_reality_mode_) {
139  writer_ptr = std::make_shared<blocker::IntraWriter<MessageT>>(new_attr);
140  } else {
141  writer_ptr = std::make_shared<Writer<MessageT>>(new_attr);
142  }
143 
144  RETURN_VAL_IF_NULL(writer_ptr, nullptr);
145  RETURN_VAL_IF(!writer_ptr->Init(), nullptr);
146  return writer_ptr;
147 }

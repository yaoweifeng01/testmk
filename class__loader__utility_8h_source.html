75 template <typename Derived, typename Base>
76 void RegisterClass(const std::string& class_name,
77  const std::string& base_class_name) {
78  AINFO << "registerclass:" << class_name << "," << base_class_name << ","
79  << GetCurLoadingLibraryName();
80 
81  utility::AbstractClassFactory<Base>* new_class_factrory_obj =
82  new utility::ClassFactory<Derived, Base>(class_name, base_class_name);
83  new_class_factrory_obj->AddOwnedClassLoader(GetCurActiveClassLoader());
84  new_class_factrory_obj->SetRelativeLibraryPath(GetCurLoadingLibraryName());
85 
86  GetClassFactoryMapMapMutex().lock();
87  ClassClassFactoryMap& factory_map =
88  GetClassFactoryMapByBaseClass(typeid(Base).name());
89  factory_map[class_name] = new_class_factrory_obj;
90  GetClassFactoryMapMapMutex().unlock();
91 }

93 template <typename Base>
94 Base* CreateClassObj(const std::string& class_name, ClassLoader* loader) {
95  GetClassFactoryMapMapMutex().lock();
96  ClassClassFactoryMap& factoryMap =
97  GetClassFactoryMapByBaseClass(typeid(Base).name());
98  AbstractClassFactory<Base>* factory = nullptr;
99  if (factoryMap.find(class_name) != factoryMap.end()) {
100  factory = dynamic_cast<utility::AbstractClassFactory<Base>*>(
101  factoryMap[class_name]);
102  }
103  GetClassFactoryMapMapMutex().unlock();
104 
105  Base* classobj = nullptr;
106  if (factory && factory->IsOwnedBy(loader)) {
107  classobj = factory->CreateObj();
108  }
109 
110  return classobj;
111 }

113 template <typename Base>
114 std::vector<std::string> GetValidClassNames(ClassLoader* loader) {
115  std::lock_guard<std::recursive_mutex> lck(GetClassFactoryMapMapMutex());
116 
117  ClassClassFactoryMap& factoryMap =
118  GetClassFactoryMapByBaseClass(typeid(Base).name());
119  std::vector<std::string> classes;
120  for (auto& class_factory : factoryMap) {
121  AbstractClassFactoryBase* factory = class_factory.second;
122  if (factory && factory->IsOwnedBy(loader)) {
123  classes.emplace_back(class_factory.first);
124  }
125  }
126 
127  return classes;

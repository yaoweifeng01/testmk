264 template <typename T>
265 void Blocker<T>::Enqueue(const MessagePtr& msg) {
266  if (attr_.capacity == 0) {
267  return;

269  std::lock_guard<std::mutex> lock(msg_mutex_);
270  published_msg_queue_.push_front(msg);

133 template <typename MessageT>
134 bool Reader<MessageT>::Init() {
135  if (init_.exchange(true)) {
136  return true;
137  }
138  std::function<void(const std::shared_ptr<MessageT>&)> func;
139  if (reader_func_ != nullptr) {
140  func = [this](const std::shared_ptr<MessageT>& msg) {
141  this->Enqueue(msg);
142  this->reader_func_(msg);
143  };
144  } else {
145  func = [this](const std::shared_ptr<MessageT>& msg) { this->Enqueue(msg); };
146  }
147  auto sched = scheduler::Instance();
148  croutine_name_ = role_attr_.node_name() + "_" + role_attr_.channel_name();
149  auto dv = std::make_shared<data::DataVisitor<MessageT>>(
150  role_attr_.channel_id(), pending_queue_size_);
151  // Using factory to wrap templates.
152  croutine::RoutineFactory factory =
153  croutine::CreateRoutineFactory<MessageT>(std::move(func), dv);
154  if (!sched->CreateTask(factory, croutine_name_)) {
155  AERROR << "Create Task Failed!";
156  init_.exchange(false);
157  return false;
158  }
159 
160  receiver_ = ReceiverManager<MessageT>::Instance()->GetReceiver(role_attr_);
161  this->role_attr_.set_id(receiver_->id().HashValue());
162  channel_manager_ =
163  service_discovery::TopologyManager::Instance()->channel_manager();
164  JoinTheTopology();
165 
166  return true;
167 }

183 template <typename MessageT>
184 void Reader<MessageT>::JoinTheTopology() {
185  // add listener
186  change_conn_ = channel_manager_->AddChangeListener(std::bind(
187  &Reader<MessageT>::OnChannelChange, this, std::placeholders::_1));
188 
189  // get peer writers
190  const std::string& channel_name = this->role_attr_.channel_name();
191  std::vector<proto::RoleAttributes> writers;
192  channel_manager_->GetWritersOfChannel(channel_name, &writers);
193  for (auto& writer : writers) {
194  receiver_->Enable(writer);
195  }
196  channel_manager_->Join(this->role_attr_, proto::RoleType::ROLE_READER,
197  message::HasSerializer<MessageT>::value);
198 }

206 template <typename MessageT>
207 void Reader<MessageT>::OnChannelChange(const proto::ChangeMsg& change_msg) {
208  if (change_msg.role_type() != proto::RoleType::ROLE_WRITER) {
209  return;
210  }
211 
212  auto& writer_attr = change_msg.role_attr();
213  if (writer_attr.channel_name() != this->role_attr_.channel_name()) {
214  return;
215  }
216 
217  auto operate_type = change_msg.operate_type();
218  if (operate_type == proto::OperateType::OPT_JOIN) {
219  receiver_->Enable(writer_attr);
220  } else {
221  receiver_->Disable(writer_attr);
222  }
223 }

235 template <typename MessageT>
236 double Reader<MessageT>::GetDelaySec() const {
237  if (latest_recv_time_sec_ < 0) {
238  return -1.0;
239  }
240  if (second_to_lastest_recv_time_sec_ < 0) {
241  return Time::Now().ToSecond() - latest_recv_time_sec_;
242  }
243  return std::max((Time::Now().ToSecond() - latest_recv_time_sec_),
244  (latest_recv_time_sec_ - second_to_lastest_recv_time_sec_));
245 }
